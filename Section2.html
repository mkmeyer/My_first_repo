number <- 2
if((number %% 15) == 0) {
  print("fizz buzz")
  } else if((number %% 5) == 0){
    print("buzz")
  } else if((number %% 3) == 0){
    print("fizz")
  } else {

  #plot with plot
  plot(0:n, dbinom(0:n, size = n, prob = prob), type = "h")

  lines(norm_x, dnorm(norm_x, 
                      mean = n*prob, 
                      sd = sqrt(n*prob*(1-prob))))
